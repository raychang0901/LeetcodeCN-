/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        //  [法一]  換成正數
        // int f = 1;
        // ListNode* temp = head;
        // while(head!=NULL){
        //     head = head->next;
        //     f++;
        // }
        // return removeNthFromBegin(temp, f-n);

        //  [法二]  雙指針
        ListNode* temp = head;
        ListNode* f = head;
        ListNode* s = head;
        while(n--){
            f = f->next;
        }
        if(f==NULL){
            return head->next;
        }
        while(f->next!=NULL){
            f = f->next;
            s = s->next;
        }
        s->next = s->next->next;
        return temp;

    }
