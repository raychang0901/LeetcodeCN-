## 题目链接

https://leetcode-cn.com/problems/count-good-numbers/

力扣5802. 统计好数字的数目

自己寫MOD的快速冪
    
---------------------------------------

```cpp
class Solution {
    int MOD = 1e9+7;
public:
    int countGoodNumbers(long long n) {
        return mypow(5, (n+1)/2)*mypow(4, n/2)%MOD;
    }
    // a^b
    long mypow(long  a, long b){
        if(b==0){
            return 1;
        }
        if(b==1){
            return a;
        }
        long long val = mypow(a, b/2)%MOD;
        if(b%2==0){
            return (val*val)%MOD;
        }
        else{
            return ((val*val)%MOD*a)%MOD;
        }
        return -1;
    }
};
```
