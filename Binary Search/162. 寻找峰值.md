力扣162. 寻找峰值

二分搜尋真的太靈活==   看了評論才想到

**if nums[mid]>nums[mid+1] 則 [mid+1 ~ N]必有峰值
    
---------------------------------------

```cpp
class Solution {
public:
    int findPeakElement(vector<int>& nums) {
        int f = 0;
        int b = nums.size()-1;
        int mid = -1;
        while(b>f){
            mid = (f+b)/2;
            if(nums[mid]>nums[mid+1]){
                b = mid;
            }
            else{
                f = mid+1;
            }
        }
        return f; 
    }
};
```
