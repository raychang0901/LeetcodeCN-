# 泛型演算法
在 435无重叠区间 遇到傳入cmp函式使sort可以更加的泛用到演算法中，將此方法紀錄下並實作

----------------------------
```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
bool cmp(int a,int b)
{
    if(!(a%2))
    {
        return a < b;
    }
    else 
        return a > b;
}
int main()
{
    vector<int> arr = {0,1,5,4,3,9};
    sort(arr.begin(), arr.end(), [](int a,int b){return a > b;});
    for(int i : arr)
    {
        cout<< i;
    }
    cout<< endl;
    sort(arr.begin(), arr.end(), cmp);
    for(int i : arr) cout<< i;
        
}

```
